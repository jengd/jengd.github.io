让GO开局就让代码有mvc的感觉

2021.08.31

让GO开局就让代码有mvc的感觉

package main

import "net/http"

// 开局就让代码有mvc的感觉
type IndexCtl struct {
	w http.ResponseWriter
}

func (this *IndexCtl) Index(req *http.Request)  {
	this.w.Write([]byte("this is index"))
}

func (this *IndexCtl) Users(req *http.Request)  {
	this.w.Write([]byte("user-lise"))
}


//转化函数
func web(f func(ctl *IndexCtl, req *http.Request)) http.Handler {
	return http.HandlerFunc(func(writer http.ResponseWriter, req *http.Request) {
		f(&IndexCtl{w: writer}, req)
	})
}

func main()  {

	router := http.NewServeMux()
	router.Handle("/", web((*IndexCtl).Index))
	router.Handle("/users", web((*IndexCtl).Users))
	http.ListenAndServe(":8080", router)
}
